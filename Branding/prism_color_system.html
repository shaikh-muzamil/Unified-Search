Rule: each result card gets a 3px left border in its integration color. The badge pill uses 15% opacity of the brand color as background, full color as text — so it reads on the dark card surface without overwhelming it.

CSS custom properties — paste into your root stylesheet

/* prism design tokens */
:root {
  /* backgrounds */
  --bg-app:       #0f0a1e;
  --bg-sidebar:   #1a1040;
  --bg-card:      #130d2a;
  --bg-input:     #1e1b4b;
  --bg-hover:     #2d1b69;
  --bg-selected:  #312e81;

  /* brand */
  --primary:      #7c3aed;
  --primary-h:    #6d28d9;
  --accent:       #a78bfa;
  --accent-dim:   #c4b5fd;

  /* text */
  --text-1:       #ffffff;
  --text-2:       #c4b5fd;
  --text-3:       #9ca3af;
  --text-ph:      #6b7280;

  /* borders */
  --border:       #374151;
  --border-focus: #7c3aed;

  /* semantic */
  --success:      #22c55e;
  --warning:      #f59e0b;
  --danger:       #ef4444;
  --info:         #06b6d4;
  --ai:           #f472b6;

  /* radii */
  --r-sm:  6px;
  --r-md:  10px;
  --r-lg:  16px;
  --r-pill: 999px;
}
